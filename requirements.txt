#   pip install -r requirements.txt
#
# WITH OPENAI SUPPORT:
#   1. Uncomment: openai>=1.0.0
#   2. Run: pip install -r requirements.txt
#   3. Set in .env: LLM_PROVIDER=openai
#
# WITH HUGGINGFACE SUPPORT:
#   1. Uncomment: huggingface-hub>=0.20.0
#   2. Run: pip install -r requirements.txt
#   3. Set in .env: LLM_PROVIDER=huggingface
#
# WITH OLLAMA SUPPORT:
#   1. Download Ollama from https://ollama.com/
#   2. Uncomment: ollama>=0.1.0
#   3. Run: pip install -r requirements.txt
#   4. Run: ollama pull llama3.1
#   5. Set in .env: LLM_PROVIDER=ollama
#
# WITH YAML CONFIG SUPPORT:
#   1. Uncomment: pyyaml>=6.0.1
#   2. Run: pip install -r requirements.txt
